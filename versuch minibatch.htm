      class KMeans():               # Input: processed dataset, Output: clustered data (kmeans, minibatch, kmeans++)
          def __init__(self, k=8, inits=300, max_iterations=300, tol=1e-3):
              self._k = k
              self._inits = inits
              self._max_iterations = max_iterations
              self._tol = tol
              
          def initialize(self, data): #setzt zufällig centroids für cluster
              indices = np.random.choice(
                  range(len(data)),
                  self._k,
                  replace=False
              )
              return data[indices]
          
          def expectation(self, data, centroids): #ordnet den centroids die Punkte zu
              centroids = np.expand_dims(centroids, axis=1)
              data_batch = np.expand_dims(data, axis=0)
              metric = np.linalg.norm(centroids - data, axis=2)
              return np.argmin(metric, axis=0)
          
          def maximization(self, data, assignments, centroids): #verschiebt die centroids zum cluster-Mittelpunkt
              new_centroids = centroids.copy()
              for idx, centroid in enumerate(centroids):
                  position = np.argwhere(assignments == idx)
                  update = data[position].mean(axis=0)
                  new_centroids[idx] = update
              return new_centroids
          
          def fit(self, data): #alles zusammen
              centroids = self.initialize(data)
              
              old_centroids = None
              for idx in range(self._max_iterations):
                  old_centroids = centroids.copy()
                  
                  assignments = self.expectation(data, centroids)
                  centroids = self.maximization(data, assignments, centroids)
                  
                  if np.linalg.norm(centroids - old_centroids) < self._tol:
                      break
      ​
              return centroids, self.expectation(data, centroids)

          def __init__(self, batch_size=128, **kwargs):
              super().__init__(**kwargs)
              self._batch_size = batch_size
              
          def create_batch(self, data): #wählt ein random batch aus den Daten aus
              data_batch = np.random.choice(
                  range(len(data)),
                  self._batch_size,
                  replace=False
              )
              return data[data_batch]
              
          def initialize(self, data): #setzt zufaellig centroids aus den Daten
              centroids = super().initialize(data)
              return centroids, np.zeros(self._k, dtype=np.int)
          
          @staticmethod
          @numba.jit(nopython=True)
          def _maximization_aux(data, assignments, centroids, centroid_count): #verschiebt die centroids richtung clustermittelpunkt mit lr
              update = centroids.copy()
              for idx, assignment in enumerate(assignments):
                  data_point = data[idx]
                  centroid_count[assignment] += 1

                  update[assignment] = update[assignment] * (1 - lr) + data_point * lr
              return update

      # linear regression using "mini-batch" gradient descent 
      # function to compute hypothesis / predictions 
      def hypothesis(X, theta): 
          return np.dot(X, theta) 
        
      # function to compute gradient of error function w.r.t. theta 
      def gradient(X, y, theta): 
          h = hypothesis(X, theta) 
          grad = np.dot(X.transpose(), (h - y)) 
          return grad 
        
      # function to compute the error for current values of theta 
      def cost(X, y, theta): 
          h = hypothesis(X, theta) 
          J = np.dot((h - y).transpose(), (h - y)) 
          J /= 2
          return J[0] 
        
      # function to create a list containing mini-batches 
      def create_mini_batches(X, y, batch_size): 
          mini_batches = [] 
          data = np.hstack((X, y)) 
          np.random.shuffle(data) 
          n_minibatches = data.shape[0] // batch_size 
          i = 0
        
          for i in range(n_minibatches + 1): 
              mini_batch = data[i * batch_size:(i + 1)*batch_size, :] 
              X_mini = mini_batch[:, :-1] 
              Y_mini = mini_batch[:, -1].reshape((-1, 1)) 
              mini_batches.append((X_mini, Y_mini)) 
          if data.shape[0] % batch_size != 0: 
              mini_batch = data[i * batch_size:data.shape[0]] 
              X_mini = mini_batch[:, :-1] 
              Y_mini = mini_batch[:, -1].reshape((-1, 1)) 
              mini_batches.append((X_mini, Y_mini)) 
          return mini_batches 
        
      # function to perform mini-batch gradient descent 
      def gradientDescent(X, y, learning_rate = 0.001, batch_size = 32): 
          theta = np.zeros((X.shape[1], 1)) 
          error_list = [] 
          max_iters = 3
          for itr in range(max_iters): 
              mini_batches = create_mini_batches(X, y, batch_size) 
              for mini_batch in mini_batches: 
                  X_mini, y_mini = mini_batch 
                  theta = theta - learning_rate * gradient(X_mini, y_mini, theta) 
                  error_list.append(cost(X_mini, y_mini, theta)) 
        
          return theta, error_list
